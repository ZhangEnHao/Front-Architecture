### 选择排序

```javascript
function selectionSort(arr) {
  var len = arr.length;
  var minIndex, temp;
  for (var i = 0; i < len - 1; i++) {
    minIndex = i;
    for (var j = i + 1; j < len; j++) {
      if (arr[j] < arr[minIndex]) {
        //寻找最小的数
        minIndex = j; //将最小数的索引保存
      }
    }
    temp = arr[i];
    arr[i] = arr[minIndex];
    arr[minIndex] = temp;
  }
  return arr;
}
```

### 二分查找算法

```javascript
function binarySearch(data, item, start, end) {
  var end = end || data.length - 1;
  var start = start || 0;
  var m = Math.floor((start + end) / 2);
  if (item === data[m]) {
    return m;
  } else if (item < data[m]) {
    return binarySearch(data, item, start, m - 1); //递归调用
  } else {
    return binarySearch(data, item, m + 1, end);
  }
  return false;
}
```

### 归并算法

```javascript
function merge(left, right) {
  var tmp = [];

  while (left.length && right.length) {
    if (left[0] < right[0]) tmp.push(left.shift());
    else tmp.push(right.shift());
  }

  return tmp.concat(left, right);
}

function mergeSort(a) {
  if (a.length === 1) return a;

  var mid = ~~(a.length / 2),
    left = a.slice(0, mid),
    right = a.slice(mid);

  return merge(mergeSort(left), mergeSort(right));
}
```

```javascript
const quickSort = (arr) => {
  let left = 0;
  let right = arr.length - 1;
  whlie(left < right) {
    while(arr[right] >= arr[0] && left < right){
      right -= 1
    }
    while(arr[left] <= arr[0] && left < right) {
      left++;
    }
    if (right === left) {
      let mid=arr[right];
      arr[right]=arr[0];
      arr[0]=mid;
      break;
    }
      // 当左指针小于右指针的位置，交换两个指针当前位置的元素
    let tem=arr[right];
    arr[right]=arr[left];
    arr[left]=tem;
  }
  return quickSort(arr.slice(0,left)).concat(arr.slice(left,right+1)).concat(quickSort(arr.slice(right+1)));
}
```

### 回文算法

```javascript
function palindrome(str) {
  var newstr = str.replace(/[^0-9a-z]/gi, "");
  newstr = newstr.toLowerCase();
  for (var i = 0, j = newstr.length - 1; i < j; i++, j--) {
    if (newstr.charAt(i) !== newstr.charAt(j)) {
      return false; //逐个字符比较，不匹配返回false
    }
  }
  return true;
}
```

### 节流函数

```javascript
function throttle(method, delay, duration) {
  var timer = null;
  var begin = new Date();
  return function () {
    var context = this,
      args = arguments;
    var current = new Date();
    clearTimeout(timer);
    if (current - begin >= duration) {
      method.apply(context, args);
      begin = current;
    } else {
      timer = setTimeout(function () {
        method.apply(context, args);
      }, delay);
    }
  };
}
```

### IndexOf 的实现

```javascript
function ArrayIndexOf(arr, value, n) {
  var i = isNaN(n) ? 0 : n; //有第三参时
  i = i < 0 ? arr.length + i : i; //第三参为负数时
  for (i; i < arr.length; i++) {
    if (arr[i] === value) {
      return i;
    }
  }
  return -1;
}
```
